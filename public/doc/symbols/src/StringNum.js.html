 26 /**
 27 * Converte de decimal para binario
 28 * @param {integer} valor, decimal
 29 * @param {integer} bits, numero de bits de retorno (default 8)
 30 * @param {boolean} espaco, se verdadeiro coloca espaço a cada 4 bits
 31 * @return {string}
 32 */
 33 function DecToBin(valor,bits,espaco) //valor = dec e bits = 8 // espaço(bool) entre 4 bits //
 34 {
 35 	if(!bits) bits =8;
 36 	var negativo = false;
 37 	if(valor<0)
 38 	{
 39 		negativo = true;
 40 		valor = valor *-1;
 41 	}
 42 	var resto = "";
 43 	do{
 44 		resto +=(valor%2);
 45 		valor = Math.floor(valor/2);
 46 	}while(valor >= 1 && resto.length <=bits);
 47 	for(var i = resto.length; i< bits;i++) resto +="0";
 48 	
 49 	resto = Reverse(resto);
 50 	if(negativo)
 51 		resto = complement2(resto);
 52 	if(espaco)
 53 		resto = ColocaEspaco(resto);
 54 	return resto;
 55 }

 62 function BinToDec(s)
 63 {
 64 	var ind = s.length-1;
 65 	var i = 0;
 66 	var dec = 0;
 67 	while((ind-i)>=0){
 68 		dec += s[ind-i]*Math.pow(2,i);
 69 		i++;
 70 	}
 71 	return dec;
 72 }

226 /**
227 * Substitui os 'k' e os 'd' pelos binario da instrução
228 * @param {string} fs: padrao do opcode; exemplo: 1111 dddd kkkk dddd
229 * @param {integer} d: numero do Rd em dec
230 * @param {integer} [k: valor imediato]
231 * @param {integer} [kb: numero de bits que se quer gerar a partir do valor de k]
232 * @param {integer} [r:  numero do Rr em dec]
233 * @param {integer} [nd: numero de bits que se quer gerar a partir do valor de d]
234 * @param {integer} [nr = numero de bits que se quer gerar a partir do valor de r]
235 * @return {string}
236 */
237 function CreateOpcode(fs,d,k,kb,r,nd,nr)
238 {
239 	if(!nd)	nd = 4;
240 	if(!nr)	nr = 4;
241 	var binD = DecToBin(d,nd); //Rd
242 	var binK = DecToBin(k,kb);//KK
243 	var binR = DecToBin(r,nr); //Rr
244 	var jk = 0;
245 	var jd = 0;
246 	var jr = 0;
247 	var ns = "";
248 	for(var i =0;i<fs.length;i++)
249 	{
250 		if(fs.substr(i,1) != " ")
251 		{
252 			if(fs.substr(i,1) == "d")
253 			{
254 				ns+= binD.substr(jd,1);
255 				jd++;
256 			}else 
257 				if(fs.substr(i,1) == "k")	
258 				{
259 					ns+= binK.substr(jk,1);
260 					jk++;
261 				}else
262 					if(fs.substr(i,1) == "r")	
263 					{
264 						ns+= binR.substr(jr,1);
265 						jr++;
266 					}else
267 						ns+=fs.substr(i,1);
268 		}else
269 			ns+=" ";
270 	
271 	}
272 	return ns;
273 }

335  function ADD(op1,op2,c)
336  {
337 	var result = "";
338 	var ult;
339 	if(!c)
340 		AVR328.C = 0; //Seta o Bit de status do Carry como 0.
341 	if(typeof(op1) != "string")
342 		op1=op1.toString();
343 	if(typeof(op2) != "string")
344 		op2=op2.toString();
345 	if(op1.length > op2.length)
346 	{
347 		var x = op1.length - op2.length;
348 		for(var i = 0; i < x; i++)
349 		{
350 			op2 = "0" + op2;
351 		}
352 	}
353 	else if(op1.length < op2.length)
354 	{
355 		var x = op2.length - op1.length;
356 		for(var i = 0; i < x; i++)
357 		{
358 			op1 = "0" + op1;
359 		}
360 	}
361 	
362 	ult = op1.length;		
363 	
364 	for(var i = ult-1; i >= 0;i--)
365 	{
366 		var j = parseInt(op1[i]);
367 		var x = parseInt(op2[i]);
368 		var r = j + x + AVR328.C;
369 		
370 		if(r > 1)
371 		{
372 			result += (r - 2);
373 			AVR328.C = 1;
374 		}
375 		else
376 		{
377 			result += r;
378 			AVR328.C = 0;
379 		}
380 	}
381 	result = Reverse(result);
382 	return result;
383 
384 }

492 		//Declara o valor para o deslocamento.

502 function AfetaFlag(sdk)
503 {
504 	var dk = BinToDec(sdk);
505 	if(dk == 0) AVR328.Z = 1; else AVR328.Z = 0;

509 function OR(s)
510 {
511 	//Converte para maiusculo e retira o comando da string incluindo os espacos.
512 	s = s.toUpperCase(); s = s.replace("OR",""); s = TrimAll(s);
513 	
514 	//Exp para validar a entrada.
515 	// exp1 = R[\d],R[\d]
516 	// exp2 = R[\d],R[\d\d]
517 	// exp3 = R[\d\d],R[\d]
518 	// exp4 = R[\d\d],R[\d\d]
519 	var exp1 = RegExp(/((([r]|[R])+([\d]))+,+(([r]|[R])+([\d])))/);
520 	var exp2 = RegExp(/((([r]|[R])+([\d]))+,+(([r]|[R])+([\d\d])))/);
521 	var exp3 = RegExp(/[R]+[\d]+[\d]+,+[R]+([\d])$/);
522 	var exp4 = RegExp(/[R]+[\d]+[\d]+,+[R]+([\d]+[\d])$/);
523 	
524 	//Declara as variaveis que receberão os numeros dos regs.
525 	var d,r;
526 	
527 	if(exp1.test(s))
528 	{
529 		d = parseInt(s.substring(1,2));
530 		r = parseInt(s.substring(4,5));
531 		if( d < 0 && r < 0)
532 			return false;
533 	}
534 	else if(exp2.test(s))
535 	{
536 		d = parseInt(s.substring(1,2));
537 		r = parseInt(s.substring(4,6));
538 		if(d < 0 && (r < 0 || r > 31))
539 			return false;			
540 	}
541 	else if(exp3.test(s))
542 	{
543 		d = parseInt(s.substring(1,3));
544 		r = parseInt(s.substring(5,6));
545 		if(r < 0 || d < 0 || d > 31)
546 			return false;
547 	}
548 	else if(exp4.test(s))
549 	{
550 		d = parseInt(s.substring(1,3));
551 		r = parseInt(s.substring(5,7));
552 		if(d < 0 || d > 31 || r < 0 || r > 31)
553 			return false;			
554 	}
555 	else
556 		return false;
557 	
558 	//Se chegou até aqui significa que a string foi informada corretamente.
559 	
560 	/*	Pega os valores de cada registrador.
561 		Depois pega o tamanho da string para ser parametro do 'for'.
562 		Nesse caso poderia ser qualquer reg. Ex.: reg[d] || reg[r],
563 		pois os valores dentro de cada registrador sempre tem o mesmo tamanha em bits.
564 		Após isso e feito um .E. entre os valores e o resultado armazenado no reg[d].
565 		Declara a variavel que receberá o .E. entre os registradores.
566 	 */
567 	 
568 	//Coloca os valores de reg[d] e reg[r] nas variaveis.
569 	var rd = AVR328.R[d];
570 	var rr = AVR328.R[r];
571 	//
572 	var ns = "";
573 	
574 	for(var i = 0; i < AVR328.R[d].length; i++)
575 	{
576 		if(Boolean(parseInt(rd[i])) || Boolean(parseInt(rr[i])))
577 			ns += "1";
578 		else
579 			ns += "0";
580 	}
581 	
582 	//Atribui o valor do .E. no reg[d].
583 	AVR328.R[d] = ns;
584 }
